def open_ngrams_as_dataframe(directory, tail: int=2):
    '''
    Loads all text files from the input directory into a dataframe,
    parsing out the date value from the filename itself.
    Also opens gzip files inside the directory.
    Tail is the lowest frequency. i.e. inlcude row if frequency > tail.
    Assumes text file format is from the sort -c command line command (i.e. read_fwf - fixed width file)
    Returns: dataframe['date', 'word', 'frequency']
    '''
    colNames = ['date', 'word', 'frequency', 'corpus']
    df = pd.DataFrame(columns=colNames)
    
    for filename in glob.glob(os.path.join(directory, '*')):
        if (os.path.splitext(filename)[1] == '.gz'):
            infile = gzip.open(filename, 'rt')
            date = pd.to_datetime(os.path.splitext(filename)[0][-14:-4]) # *.txt.gz
            corpus = os.path.splitext(filename)[0][-16:-15]  # alpha beta symbol
        else:
            infile = open(filename, 'r')
            date = pd.to_datetime(os.path.splitext(filename)[0][-10:]) # *.txt
            corpus = os.path.splitext(filename)[0][-12:-11]  # alpha beta symbol

        with infile:
            lines = [l.strip().split(' ') for l in infile.read().splitlines()]  # split all words on line
            lines = [[date, str(' '.join(l[1:])), int(l[0]), corpus] for l in lines] # re-join n+1 words in each line
            
            data = pd.DataFrame(lines, columns=colNames)
            data = data[data['frequency'] > tail]  # filter low frequency words
            df = df.append(data)
    
    # set index and return
    df['word'] = df['word'].apply(str)
    df = df.set_index('date').sort_index()
    return df
